// TPSS Stage 0 Screener — Apps Script
// Deploy: Web App · Execute as: Me · Access: Anyone
// ─────────────────────────────────────────────────────────────

const SECRET_KEY  = "tpss-2026-awais";   // ← your screener secret key
const GEMINI_KEY  = "AIza_REPLACE_ME";   // ← paste your Gemini API key here (free at aistudio.google.com)
const RESULTS_SHEET = "Stage 0 Results";

// ── READ: loads applicants into the screener tool ──────────────
function doGet(e) {
  const key = e.parameter.key;
  const cb  = e.parameter.callback;   // JSONP callback name

  if (key !== SECRET_KEY) {
    return respond({ error: "Unauthorized" }, cb);
  }

  // ── AI SCREENING (action=screen) — runs Gemini server-side ──
  if (e.parameter.action === "screen") {
    const prompt = e.parameter.prompt || "";
    return respond(screenWithGemini(prompt), cb);
  }

  const sheetName = e.parameter.sheet || "Sheet1";
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    return respond({ error: "Sheet '" + sheetName + "' not found" }, cb);
  }

  const raw = sheet.getDataRange().getValues();
  if (raw.length < 4) {
    return respond({ rows: [], submittedIds: [], total: 0 }, cb);
  }

  const headers = raw[2].map(h => String(h).trim());  // row 3 = index 2
  const rows = raw.slice(3)                            // data starts row 4
    .filter(r => r.some(c => c !== "" && c !== null && c !== undefined))
    .map(r => {
      const obj = {};
      headers.forEach((h, i) => {
        const val = r[i];
        obj[h] = (val !== null && val !== undefined) ? String(val).trim() : "";
      });
      return obj;
    });

  const submittedIds = getSubmittedIds();
  return respond({ rows: rows, submittedIds: submittedIds, total: rows.length }, cb);
}

// ── POST: handles both AI screening and saving results ─────────
function doPost(e) {
  let body;
  try { body = JSON.parse(e.postData.contents); }
  catch (err) { return respond({ error: "Invalid JSON: " + err.toString() }, null); }

  if (body.key !== SECRET_KEY) {
    return respond({ error: "Unauthorized" }, null);
  }

  // ── AI SCREENING (action=screen) ────────────────────────────
  if (body.action === "screen") {
    return respond(screenWithGemini(body.prompt), null);
  }

  // ── SAVE RESULT (default) ───────────────────────────────────
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(RESULTS_SHEET);

  if (!sh) {
    sh = ss.insertSheet(RESULTS_SHEET);
    const headerRow = [
      "Timestamp","Venture Name","Applicant","Country","Sector",
      "App Stage","Stream","Assigned To",
      "Gate: Venture Stage","Gate: Ummah Align","Gate: Compliance",
      "Gate: Team","Gate: Viability","Gate: Availability",
      "AI Score","Human Score","Score Delta",
      "Rating: Ummah","Rating: Stage","Rating: Team","Rating: Viability",
      "Verdict","Notes","Edited Gates","Submitted At"
    ];
    sh.appendRow(headerRow);
    sh.getRange(1,1,1,headerRow.length)
      .setBackground("#1a2e5a")
      .setFontColor("#ffffff")
      .setFontWeight("bold");
    sh.setFrozenRows(1);
  }

  const d = body;
  sh.appendRow([
    new Date().toISOString(),
    d.ventureName   || "",
    d.applicantName || "",
    d.country       || "",
    d.sector        || "",
    d.ventureStage  || "",
    d.stream        || "",
    d.assignedTo    || "Unassigned",
    d.gates ? d.gates.stage        : "",
    d.gates ? d.gates.ummah        : "",
    d.gates ? d.gates.compliance   : "",
    d.gates ? d.gates.team         : "",
    d.gates ? d.gates.viability    : "",
    d.gates ? d.gates.availability : "",
    d.aiScore    || 0,
    d.humanScore || 0,
    (d.aiScore || 0) - (d.humanScore || 0),
    d.ratings ? d.ratings.ummah      : 0,
    d.ratings ? d.ratings.stage      : 0,
    d.ratings ? d.ratings.team       : 0,
    d.ratings ? d.ratings.viability  : 0,
    d.verdict    || "",
    d.notes      || "",
    Array.isArray(d.editedGates) ? d.editedGates.join(", ") : "",
    d.submittedAt || new Date().toISOString()
  ]);

  return respond({ status: "ok", saved: d.ventureName || "unknown" }, null);
}

// ── GEMINI SCREENING (server-side via UrlFetchApp, no CORS) ────
function screenWithGemini(prompt) {
  try {
    const url = "https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=" + GEMINI_KEY;
    const response = UrlFetchApp.fetch(url, {
      method: "post",
      contentType: "application/json",
      payload: JSON.stringify({
        contents: [{ parts: [{ text: prompt }] }],
        generationConfig: { temperature: 0.2, maxOutputTokens: 1000 }
      }),
      muteHttpExceptions: true
    });

    const data = JSON.parse(response.getContentText());
    if (data.error) throw new Error(data.error.message);
    const raw = data.candidates[0].content.parts[0].text || "";
    const parsed = JSON.parse(raw.replace(/```json|```/g, "").trim());
    return { ok: true, result: parsed };
  } catch(err) {
    return { ok: false, error: err.toString() };
  }
}

// ── HELPERS ───────────────────────────────────────────────────
function getSubmittedIds() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(RESULTS_SHEET);
  if (!sh || sh.getLastRow() < 2) return [];
  const data = sh.getRange(2, 2, sh.getLastRow() - 1, 2).getValues();
  return data
    .map(r => (String(r[0]) + "|" + String(r[1])).toLowerCase().trim())
    .filter(Boolean);
}

// respond() — always includes CORS headers so browsers can POST from any origin
function respond(obj, callbackName) {
  const json = JSON.stringify(obj);
  if (callbackName) {
    return ContentService
      .createTextOutput(callbackName + "(" + json + ");")
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  // For POST responses, Apps Script doesn't support custom headers directly,
  // so we use JSONP-style wrapping with a fixed callback name the client expects.
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

// doOptions() handles the browser's CORS preflight OPTIONS request.
// Without this, every POST is blocked before it even reaches doPost().
function doOptions(e) {
  return ContentService
    .createTextOutput("")
    .setMimeType(ContentService.MimeType.TEXT);
}
